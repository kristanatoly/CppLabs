//Дан массив натуральных чисел n размерности m (m вводится с клавиатуры) и
//массив действительных чисел x той же размерности.
//Вычислить (x1*n1+…+xm*nm)/(n1+…+nm).

#include <iostream>

using namespace std;

int main()
{
	int m;
	cin >> m;
	
	int *n = new int[m];
	double *x = new double[m];
	
	for (int i = 0; i < m; i++)
		cin >> n[i];
	
	for (int i = 0; i < m; i++)
		cin >> x[i];
	
	double s1 = 0;
	double s2 = 0;
	
	for (int i = 0; i < m; i++)
	{
		s1 = s1 + x[i] * n[i];
		s2 = s2 + n[i];
	}
	
	double s;
	s = s1 / s2;
	cout << s;

	return 0;
}
