//ƒан массив размерности n. ≈сли отрицательные и положительные элементы в
//массиве чередуютс€ (+ - + - + Е или - + - + - + Е) заменить максимальные
//элементы минимальными, иначе Ц наоборот.

#include <iostream>
using namespace std;

int main()
{
    int n;
    cin >> n;
    int *a = new int[n];
    
    bool check = 1;
    cin >> a[0];
    int max = a[0], min = a[0];
    
    for (int i = 1; i < n; i++)
    {
        cin >> a[i];
        
        if (a[i] > max)
            max = a[i];
        
        if (a[i] < min)
            min = a[i];
 
        if (a[i] >= 0)
        {
            if (a[i - 1] >= 0)
                check = 0;
        }
        
        if (a[i] < 0)
        {
            if (a[i - 1] < 0)
                check = 0;
        }
    }
    for (int i = 0; i < n; i++)
    {
        if (check == 1)
        {
            if (a[i] == max)
                a[i] = min;
        }
        else
        {
            if (a[i] == min)
                a[i] = max;
        }
    }
    for (int i = 0; i < n; i++)
        cout << a[i] << " ";
   
    return 0;
}
